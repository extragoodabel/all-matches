async function resolveMatchId(incomingId: number, userId: number) {
  // First try incomingId as matchId
  const matches = await storage.getMatches(userId)
  let match = matches.find(m => m.id === incomingId)
  if (match) return { match, matchId: match.id }

  // Fallback: treat incomingId as profileId
  match = matches.find(m => m.profileId === incomingId)
  if (match) return { match, matchId: match.id }

  return null
}
